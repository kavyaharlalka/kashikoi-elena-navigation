31def get_best_path(algorithm_id, graph, source_nearest_nodes, destination_nearest_nodes, path_percentage, minimize_elevation_gain: bool): 

32 """ Get the best path between two nodes as per the given algorithm. 

33 Parameters: 

34 algorithm_id -> Id of the algorithm to be used to get the best path 

35 graph -> The graph of the map containing the nodes 

36 source_nearest_nodes -> Nodes nearest to the source 

37 destination_nearest_nodes -> Nodes nearest to the destination 

38 path_percentage -> Percentage of the shortest path 

39 minimize_elevation_gain -> Will minimize elevation gain if set to true 

40 Returns: 

41 Array of nodes and array of coordinates for the best path 

42 """ 

43 # algorithms = ["Dijkstra", "Bidirectional Dijkstra", "A *", "Bellman - Ford", "Goldberg - Radzik", "Johnson", "Floyd - Warshall"] 

44 assert algorithm_id in [e.value for e in Algorithms], "Invalid Algorithm ID" 

45 assert graph is not None, "Invalid Location" 

46 assert source_nearest_nodes is not None, "Invalid Source" 

47 assert destination_nearest_nodes is not None, "Invalid Destination" 

48 assert path_percentage in range(100, 501), "Invalid Path Percentage" 

49 

50 print(source_nearest_nodes) 

51 print(destination_nearest_nodes) 

52 

53 nx_graph = networkx.Graph(graph) 

54 shortest_path_distance = networkx.dijkstra_path_length(nx_graph, source=source_nearest_nodes, target=destination_nearest_nodes, weight='length') 

55 max_distance = (path_percentage * shortest_path_distance) / 100.0 

56 

57 is_algorithm_with_not_supporting_none = algorithm_id == Algorithms.BELLMANFORD.value or algorithm_id == Algorithms.GOLDBERG_RADZIK.value 

58 destination_node = graph.nodes[destination_nearest_nodes] 

59 def custom_weight_func(u, v, data): 

60 elevation_gain = abs(nx_graph.nodes[v]['elevation'] - nx_graph.nodes[u]['elevation']) 

61 

62 current_node = nx_graph.nodes[v] 

63 distance_from_destination = get_distance_from_destination(destination_node[constants.COORDINATES_Y], 

64 destination_node[constants.COORDINATES_X], 

65 current_node[constants.COORDINATES_Y], 

66 current_node[constants.COORDINATES_X]) 

67 

68 if distance_from_destination > max_distance or elevation_gain < 0: # pragma: no cover 

69 return sys.maxsize if is_algorithm_with_not_supporting_none else None # Ignore paths that exceed the max_distance 

70 

71 # You can customize how elevation gain is prioritized (minimized or maximized) 

72 return 1.0/elevation_gain if (not minimize_elevation_gain and not elevation_gain == 0) else elevation_gain 

73 

74 best_path = [] 

75 # Dijkstra 

76 if algorithm_id == Algorithms.DIJKSTRA.value: 

77 best_path = networkx.dijkstra_path(nx_graph, source=source_nearest_nodes, target=destination_nearest_nodes, weight=custom_weight_func) 

78 # Bi-Directional Dijkstra 

79 elif algorithm_id == Algorithms.BI_DIJKSTRA.value: 

80 best_path = networkx.bidirectional_dijkstra(nx_graph, source_nearest_nodes, destination_nearest_nodes, weight=custom_weight_func)[1] 

81 # A * 

82 elif algorithm_id == Algorithms.ASTAR.value: 

83 best_path = networkx.astar_path(nx_graph, source_nearest_nodes, destination_nearest_nodes, weight=custom_weight_func) 

84 # Bellman-Ford 

85 elif algorithm_id == Algorithms.BELLMANFORD.value: 

86 best_path = networkx.bellman_ford_path(nx_graph, source_nearest_nodes, destination_nearest_nodes, weight=custom_weight_func) 

87 # Goldberg-Radzik 

88 elif algorithm_id == Algorithms.GOLDBERG_RADZIK.value: 

89 best_path = [destination_nearest_nodes] 

90 # goldberg_radzik algorithm 

91 predecessors, distances = networkx.goldberg_radzik(networkx.DiGraph(graph), source_nearest_nodes, weight='elevation') 

92 while best_path[-1] != source_nearest_nodes: 

93 best_path.append(predecessors[best_path[-1]]) 

94 # Floyd-Warshall 

95 elif algorithm_id == Algorithms.FLOYD_WARSHALL.value: # pragma: no cover 

96 best_path = [destination_nearest_nodes] 

97 if not source_nearest_nodes == destination_nearest_nodes: 

98 predecessors, distance = networkx.floyd_warshall_predecessor_and_distance(nx_graph, weight='length') 

99 best_path = networkx.reconstruct_path(source_nearest_nodes, destination_nearest_nodes, predecessors) 

100 

101 return {"nodes": best_path, "coordinates": get_coordinates_from_nodes(graph, best_path)}

133def get_distance_from_destination(destination_latitude, destination_longitude, node_latitude, node_longitude): 

134 """ Get the spherical distance from destination to the given node using the Haversine formula 

135 Parameters: 

136 destination_latitude -> Latitude coordinate of the destination 

137 destination_longitude -> Longitude coordinate of the destination 

138 node_latitude -> Latitude coordinate of the node to which the distance is to be calculated 

139 node_longitude -> Longitude coordinate of the node to which the distance is to be calculated 

140 Returns: 

141 Spherical distance from the destination to the given node 

142 """ 

143 # Convert degrees to radians 

144 coordinates = destination_latitude, destination_longitude, node_latitude, node_longitude 

145 phi_1, lambda_1, phi_2, lambda_2 = [ 

146 np.radians(c) for c in coordinates 

147 ] 

148 

149 # Haversine formula 

150 multiplier = 6371008.8 

151 haversine_result = (np.square(np.sin((phi_2 - phi_1) / 2)) + np.cos(phi_1) * np.cos(phi_2) * 

152 np.square(np.sin((lambda_2 - lambda_1) / 2))) 

153 distance = 2 * multiplier * np.arcsin(np.sqrt(haversine_result)) 

154 

155 return distance

158def calculate_and_get_elevation(graph, path_nodes, elevation_mode, return_individual_costs=False): 

159 """ Computes the elevation cost of a route. 

160 Parameters: 

161 graph -> The graph of the map containing the nodes 

162 path_nodes -> Array of Node IDs

168 """ 

169 assert elevation_mode in constants.ELEVATION_MODES, "Invalid elevation mode"

171 total_cost = 0

173 for i in range(len(path_nodes) - 1):

180def get_cost_between_nodes(graph, node_1, node_2, elevation_mode="vanilla"): 

181 """ Computes the elevation cost between two nodes. 

182 Parameters: 

183 graph -> The graph of the map containing the nodes 

184 node_1 -> The first node 

185 node_2 -> The second node 

186 elevation_mode -> Mode of elevation 

187 Returns: 

188 Cost between the two nodes 

189 """ 

190 assert node_1 is not None and node_2 is not None, "Invalid node inputs" 

191 assert elevation_mode in constants.ELEVATION_MODES, "Invalid elevation mode" 

192 

193 if elevation_mode == "gain": 

194 return max(0.0, graph.nodes[node_2][constants.KEY_ELEVATION] - graph.nodes[node_1][constants.KEY_ELEVATION]) 

195 else: 

196 if graph.nodes[node_2][constants.KEY_ELEVATION] - graph.nodes[node_1][constants.KEY_ELEVATION] > 0: 

197 return 0.0 

198 else: 

199 return graph.nodes[node_1][constants.KEY_ELEVATION] - graph.nodes[node_2][constants.KEY_ELEVATION]

202def get_route_edge_attributes(graph, path_nodes, attribute_name): 

203 """ 

204 Reference - https://osmnx.readthedocs.io/en/stable/osmnx.html#osmnx.utils_graph.get_route_edge_attributes 

205 Get a list of attribute values for each edge in a path. 

206 Parameters: 

207 graph -> input graph 

208 path_nodes -> list of nodes IDs constituting the path 

209 attribute_name -> the name of the attribute to get the value of for each edge. 

210 Returns: 

211 attribute_values -> list of edge attribute values 

212 """ 

213 minimize_key = "length" 

214 attribute_values = [] 

215 for u, v in zip(path_nodes[:-1], path_nodes[1:]): 

216 edge_data = graph.get_edge_data(u, v) 

217 if graph.get_edge_data(u, v) is None: 

218 continue 

219 else: 

220 # if there are parallel edges between two nodes, select the one with the lowest value of minimize_key 

221 data = min(edge_data.values(), key=lambda x: x[minimize_key]) 

222 attribute_value = data[attribute_name] 

223 attribute_values.append(attribute_value) 

224 return attribute_values
